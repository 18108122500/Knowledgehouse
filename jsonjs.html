//定义JSON对象
var jsonObject = {"name":"albert","age":18};
alert(jsonObject.name);
var jsonObject2 = new Object();
jsonObject2.name="albert2";
jsonObject2.age=25;
alert(jsonObject2.age);
alert(jsonObject);//输出 object object
//json 对象转json string
var jsonStr = JSON.stringify(jsonObject);//输出 {"name":"albert","age":18}
alert(jsonStr);

//定义json 字符串
var jsonstr2 = '{"name":"albert","age":35}';
alert(jsonstr2);

//json 字符串转json对象
var jsonObject3 = JSON.parse(jsonstr2);
alert(jsonObject3.age);//35
